$scope.XssProtection = function($string) {
    //var filteredMsg = "<pre>";
	var filteredMsg = '';
    var inHashtag = false;
    for (var i = 0; i < $string.length; ++i) {
		var ch = $string.charAt(i);
		if (ch == '<') {
	    	filteredMsg+="&lt;";
		} else if (ch == '>') {
	    	filteredMsg+="&gt;";
		} else if (ch == '\"') {
	    	filteredMsg+="&quot;";
		} else if (ch == '#' && !inHashtag) {
	    	inHashtag = true;
	    	filteredMsg+="<strong>"+ch;
		} else if (inHashtag && (ch == ' ' || ch == '\n')) {
	    	inHashtag = false;
	    	filteredMsg+="</strong>"+ch;
		} else {
	    	filteredMsg+=ch;
		}
    }
    //filteredMsg+="</pre>";
    return filteredMsg;
};
